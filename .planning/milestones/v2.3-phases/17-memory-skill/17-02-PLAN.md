        handler::server::tool::ToolRouter,

    // --- Parameter types ---
    #[derive(Debug, Deserialize, JsonSchema)]
    pub struct StoreParams {

        pub content: String,
        #[schemars(description = "Comma-separated tags for categorization")]
        pub tags: Option<String>,
    }

    #[derive(Debug, Deserialize, JsonSchema)]

        #[schemars(description = "Tag or keyword to search by")]

    pub struct SearchParams {
        #[schemars(description = "Full-text search query")]
        pub query: String,
    }

    #[derive(Debug, Deserialize, JsonSchema)]
    pub struct ForgetParams {
        #[schemars(description = "Memory ID to soft-delete")]
        pub id: i64,
    }

    // --- Server struct ---
    #[derive(Clone)]
    pub struct MemoryServer {
        tool_router: ToolRouter<Self>,
        conn: std::sync::Arc<Mutex<rusqlite::Connection>>,
        agent_name: String,
    }

    The server wraps the Connection in `Arc<Mutex<Connection>>` because rmcp tool handlers take `&self` and Connection is not Sync. The `agent_name` comes from `std::env::var("RC_AGENT_NAME").unwrap_or_else(|_| "unknown".to_string())`.

    Implement `#[tool_router] impl MemoryServer` with 4 tool methods:

    **`store` tool** (description: "Store a memory. Content is scanned for prompt injection. Returns memory ID."):
    - Lock mutex, call `rightclaw::memory::store::store_memory(&conn, &params.content, params.tags.as_deref(), &self.agent_name, "mcp:store")`
    - On `Ok(id)`: return `CallToolResult::success(vec![Content::text(format!("stored memory id={id}"))])`
    - On `Err(MemoryError::InjectionDetected)`: return `Err(McpError::invalid_params("content rejected: possible prompt injection detected", None))`
    - On other Err: return `Err(McpError::internal_error(format!("{e:#}"), None))`

    **`recall` tool** (description: "Look up memories by tag or keyword. Returns matching active memories."):

    - Return `CallToolResult::success(vec![Content::text(json_output)])`

    **`search` tool** (description: "Full-text search memories using FTS5. Returns BM25-ranked results."):

    **1. Create `crates/rightclaw/src/codegen/mcp_config.rs`:**

    ```rust
    use std::path::Path;

    /// Merge the `rightmemory` MCP server entry into an agent's `.mcp.json`.
    ///
    /// - If `.mcp.json` exists, reads it, parses as JSON object, injects/updates
    ///   `mcpServers.rightmemory` key, writes back.
    /// - If `.mcp.json` does not exist, creates it with just the rightmemory entry.
    /// - Preserves all other keys in the existing JSON (non-destructive merge per D-05).
    pub fn generate_mcp_config(agent_path: &Path) -> miette::Result<()> {
        let mcp_path = agent_path.join(".mcp.json");

        let mut root: serde_json::Value = if mcp_path.exists() {
            let content = std::fs::read_to_string(&mcp_path)
                .map_err(|e| miette::miette!("failed to read .mcp.json: {e:#}"))?;
            serde_json::from_str(&content)
                .map_err(|e| miette::miette!("failed to parse .mcp.json: {e:#}"))?
        } else {
            serde_json::json!({})
        };

        let obj = root.as_object_mut()
            .ok_or_else(|| miette::miette!(".mcp.json root is not a JSON object"))?;

        // Ensure mcpServers key exists as object
        if !obj.contains_key("mcpServers") {
            obj.insert("mcpServers".to_string(), serde_json::json!({}));
        }

        let servers = obj.get_mut("mcpServers")
            .and_then(|v| v.as_object_mut())
            .ok_or_else(|| miette::miette!(".mcp.json mcpServers is not a JSON object"))?;

        // Insert/update the rightmemory entry (per D-05)
        servers.insert("rightmemory".to_string(), serde_json::json!({
            "command": "rightclaw",
            "args": ["memory-server"],
            "env": {}
        }));

        let output = serde_json::to_string_pretty(&root)
            .map_err(|e| miette::miette!("failed to serialize .mcp.json: {e:#}"))?;
        std::fs::write(&mcp_path, output)
            .map_err(|e| miette::miette!("failed to write .mcp.json: {e:#}"))?;

        Ok(())
    }
    ```
