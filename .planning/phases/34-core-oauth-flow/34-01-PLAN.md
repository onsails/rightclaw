       ```rust
           pub token: String,

       }

       pub fn read_global_config(home: &Path) -> miette::Result<GlobalConfig> {
           let path = home.join("config.yaml");
           if !path.exists() {
               return Ok(GlobalConfig::default());
           }
           let content = std::fs::read_to_string(&path)
               .map_err(|e| miette::miette!("read config.yaml: {e:#}"))?;
           // Parse with serde-saphyr (Deserialize impl needed)
           // Since serde-saphyr is deserialize-only, implement manual Deserialize
           // or parse the YAML structure manually via serde_json roundtrip
           // Simplest: use serde-saphyr's from_str with a helper struct
           ...
       }

       pub fn write_global_config(home: &Path, config: &GlobalConfig) -> miette::Result<()> {
           let path = home.join("config.yaml");
           // serde-saphyr cannot serialize — write YAML manually (schema is tiny)
           let mut content = String::new();
           if let Some(ref tunnel) = config.tunnel {
               content.push_str("tunnel:\n");
               content.push_str(&format!("  token: \"{}\"\n", tunnel.token));
               content.push_str(&format!("  url: \"{}\"\n", tunnel.url));
           }
           std::fs::write(&path, &content)
               .map_err(|e| miette::miette!("write config.yaml: {e:#}"))?;
           Ok(())
       }
       ```
