       use std::collections::HashMap;
       use std::path::{Path, PathBuf};
       use std::sync::Arc;
       use std::time::{Duration, Instant};
       use tokio::sync::Mutex;
       use axum::{Router, extract::{Query, Path as AxumPath, State}, http::StatusCode, response::Html, routing::get};
       use rightclaw::mcp::oauth::{PendingAuth, verify_state, exchange_token, TokenResponse};
       use rightclaw::mcp::credentials::{write_credential, mcp_oauth_key, CredentialToken};

       pub type PendingAuthMap = Arc<Mutex<HashMap<String, PendingAuth>>>;

       #[derive(Clone)]
       pub struct OAuthCallbackState {
           pub pending_auth: PendingAuthMap,
           pub credentials_path: PathBuf,
           pub agent_name: String,
           pub pc_port: u16,
           /// Telegram bot for sending completion notification
           pub bot: teloxide::Bot,
           /// Chat IDs to notify on OAuth completion
           pub notify_chat_ids: Vec<i64>,
       }

       #[derive(serde::Deserialize)]
       struct CallbackParams {
           code: Option<String>,
           state: Option<String>,
           error: Option<String>,
           error_description: Option<String>,
       }

       async fn handle_oauth_callback(
           AxumPath(agent_name): AxumPath<String>,
           Query(params): Query<CallbackParams>,
           State(state): State<Arc<OAuthCallbackState>>,
       ) -> Result<Html<String>, StatusCode> {
           // If OAuth error from AS, return it
           if let Some(err) = params.error {
               let desc = params.error_description.unwrap_or_default();
               tracing::warn!(agent = %agent_name, "OAuth error: {err}: {desc}");
               return Ok(Html(format!("<html><body>OAuth error: {err} — {desc}</body></html>")));
           }

           let received_state = params.state.ok_or(StatusCode::BAD_REQUEST)?;
           let code = params.code.ok_or(StatusCode::BAD_REQUEST)?;

           // Constant-time state lookup + verify
           let pending = {
               let guard = state.pending_auth.lock().await;
               guard.get(&received_state)
                   .filter(|p| verify_state(&p.state, &received_state))
                   .cloned()
           };
           let pending = pending.ok_or_else(|| {
               tracing::warn!("OAuth callback with unknown/invalid state");
               StatusCode::BAD_REQUEST
           })?;

           // Consume (one-shot)
           state.pending_auth.lock().await.remove(&received_state);

           // Spawn token exchange + credential write + restart in background
           let state_clone = Arc::clone(&state);
           let pending_clone = pending.clone();
           tokio::spawn(async move {
               match complete_oauth_flow(&state_clone, &pending_clone, &code).await {
                   Ok(()) => {
                       let msg = format!("{} authenticated — agent {} restarting", pending_clone.server_name, state_clone.agent_name);
                       tracing::info!("{msg}");
                       notify_telegram(&state_clone.bot, &state_clone.notify_chat_ids, &msg).await;
                   }
                   Err(e) => {
                       let msg = format!("OAuth token exchange failed for {}: {e:#}", pending_clone.server_name);
                       tracing::error!("{msg}");
                       notify_telegram(&state_clone.bot, &state_clone.notify_chat_ids, &msg).await;
                   }
               }
           });

           Ok(Html("<html><body>Authentication successful! You can close this window.</body></html>".to_string()))
       }

       async fn complete_oauth_flow(state: &OAuthCallbackState, pending: &PendingAuth, code: &str) -> miette::Result<()> {
           let client = reqwest::Client::new();
           let token_resp = exchange_token(
               &client, &pending.token_endpoint, code,
               &pending.redirect_uri, &pending.client_id,
               pending.client_secret.as_deref(), &pending.code_verifier,
           ).await.map_err(|e| miette::miette!("token exchange: {e:#}"))?;

           // Build CredentialToken
           let expires_at = token_resp.expires_in.map(|secs| {
               std::time::SystemTime::now()
                   .duration_since(std::time::UNIX_EPOCH)
                   .unwrap_or_default()
                   .as_secs() + secs
           });
           let cred = CredentialToken {
               access_token: token_resp.access_token,
               token_type: token_resp.token_type,
               expires_at,
               refresh_token: token_resp.refresh_token,
               scope: token_resp.scope,
           };

           // Write credential under CC-expected key
           let key = mcp_oauth_key(&pending.server_name, "http", &pending.server_url);
           write_credential(&state.credentials_path, &key, &cred)
               .map_err(|e| miette::miette!("write credential: {e:#}"))?;

           // Restart agent via process-compose REST API
           let pc = rightclaw::runtime::pc_client::PcClient::new(state.pc_port)?;
           let process_name = format!("{}-bot", state.agent_name);
           pc.restart_process(&process_name).await?;

           Ok(())
       }

       async fn notify_telegram(bot: &teloxide::Bot, chat_ids: &[i64], msg: &str) {
           use teloxide::requests::Requester as _;
           for &id in chat_ids {
               if let Err(e) = bot.send_message(teloxide::types::ChatId(id), msg).await {
                   tracing::warn!("failed to notify chat {id}: {e:#}");
               }
           }
       }
       ```

       pub async fn run_pending_auth_cleanup(pending_auth: PendingAuthMap) {
           let mut interval = tokio::time::interval(Duration::from_secs(60));
           loop {
               interval.tick().await;
               let mut guard = pending_auth.lock().await;
               let before = guard.len();
               guard.retain(|_, v| v.created_at.elapsed() < Duration::from_secs(600)); // 10 min
               let removed = before - guard.len();
               if removed > 0 {
                   tracing::debug!("cleaned up {removed} stale PendingAuth entries");
               }
           }
       }

           bot: Bot, msg: Message, server_name: &str,
           agent_dir: &Path, pending_auth: PendingAuthMap, home: &Path,
       ) -> miette::Result<()> {

           let mcp_content = std::fs::read_to_string(&mcp_path)
               .map_err(|e| miette::miette!("read .mcp.json: {e:#}"))?;
           let mcp_json: serde_json::Value = serde_json::from_str(&mcp_content)
               .map_err(|e| miette::miette!("parse .mcp.json: {e:#}"))?;
           let server = mcp_json.get("mcpServers").and_then(|s| s.get(server_name))
               .ok_or_else(|| miette::miette!("server '{server_name}' not found in .mcp.json"))?;
           let server_url = server.get("url").and_then(|v| v.as_str())
               .ok_or_else(|| miette::miette!("server '{server_name}' has no url field"))?;
           let static_client_id = server.get("clientId").and_then(|v| v.as_str());

           // 2. Read tunnel config
           let global_config = rightclaw::config::read_global_config(home)?;
           let tunnel = global_config.tunnel.as_ref()
               .ok_or_else(|| miette::miette!("tunnel not configured — run `rightclaw init --tunnel-token TOKEN --tunnel-url URL`"))?;

           // 3. Check cloudflared binary (OAUTH-04)
           if which::which("cloudflared").is_err() {
               bot.send_message(msg.chat.id, "Error: cloudflared binary not found in PATH. Install cloudflared first.").await?;
               return Ok(());
           }

           // 4. AS discovery (OAUTH-02)
           let http_client = reqwest::Client::new();
           bot.send_message(msg.chat.id, format!("Discovering OAuth endpoints for {server_name}...")).await?;
           let metadata = rightclaw::mcp::oauth::discover_as(&http_client, server_url).await
               .map_err(|e| miette::miette!("AS discovery failed: {e:#}"))?;

           // 5. DCR or static clientId (OAUTH-03)
           let (client_id, client_secret) = {
               let agent_name = agent_dir.file_name().unwrap().to_string_lossy();
               let redirect_uri = format!("https://{}/oauth/{}/callback", tunnel.url, agent_name);
               rightclaw::mcp::oauth::register_client_or_fallback(
                   &http_client, &metadata, static_client_id, &redirect_uri,
               ).await.map_err(|e| miette::miette!("client registration failed: {e:#}"))?
           };

           // 6. Generate PKCE + state
           let (code_verifier, code_challenge) = rightclaw::mcp::oauth::generate_pkce();
           let state = rightclaw::mcp::oauth::generate_state();
           let agent_name = agent_dir.file_name().unwrap().to_string_lossy().to_string();
           let redirect_uri = format!("https://{}/oauth/{}/callback", tunnel.url, agent_name);

           // 7. Tunnel healthcheck (OAUTH-05) — verify tunnel URL is reachable
           let healthcheck_url = format!("https://{}/oauth/{}/callback", tunnel.url, agent_name);
           match http_client.get(&healthcheck_url).timeout(std::time::Duration::from_secs(5)).send().await {
               Ok(_) => {} // Any response (even 404) means tunnel is reachable
               Err(e) => {
                   bot.send_message(msg.chat.id, format!("Tunnel healthcheck failed: {e:#}\nIs cloudflared running?")).await?;
                   return Ok(());
               }
           }

           // 8. Store PendingAuth (OAUTH-06)
           let pending = rightclaw::mcp::oauth::PendingAuth {
               server_name: server_name.to_string(),
               server_url: server_url.to_string(),
               code_verifier,
               state: state.clone(),
               token_endpoint: metadata.token_endpoint.clone(),
               client_id: client_id.clone(),
               client_secret,
               redirect_uri: redirect_uri.clone(),
               created_at: std::time::Instant::now(),
           };
           pending_auth.lock().await.insert(state.clone(), pending);

           // 9. Build and send auth URL
           let auth_url = rightclaw::mcp::oauth::build_auth_url(
               &metadata, &client_id, &redirect_uri, &state, &code_challenge, None,
           );
           bot.send_message(msg.chat.id, format!("Authenticate {server_name}:\n\n{auth_url}")).await?;
           Ok(())
       }
       ```

           // Parse: expecting "<name> <url>" or "<name> <url> <clientId>"
           let parts: Vec<&str> = config_str.split_whitespace().collect();
           if parts.len() < 2 {
               bot.send_message(msg.chat.id, "Usage: /mcp add <name> <url> [clientId]").await?;
               return Ok(());
           }
           let name = parts[0];
           let url = parts[1];
           let client_id = parts.get(2).copied();

           let mcp_path = agent_dir.join(".mcp.json");
           let mut root: serde_json::Value = if mcp_path.exists() {
               serde_json::from_str(&std::fs::read_to_string(&mcp_path)?)?
           } else {
               serde_json::json!({})
           };
           let servers = root.as_object_mut().unwrap()
               .entry("mcpServers").or_insert_with(|| serde_json::json!({}))
               .as_object_mut().ok_or_else(|| miette::miette!("mcpServers is not object"))?;

           let mut entry = serde_json::json!({"url": url});
           if let Some(cid) = client_id {
               entry.as_object_mut().unwrap().insert("clientId".to_string(), serde_json::json!(cid));
           }
           servers.insert(name.to_string(), entry);

           std::fs::write(&mcp_path, serde_json::to_string_pretty(&root)?)?;
           bot.send_message(msg.chat.id, format!("Added MCP server: {name}")).await?;
           Ok(())
       }

               bot.send_message(msg.chat.id, format!("Removed MCP server: {server_name}")).await?;
           } else {
               bot.send_message(msg.chat.id, format!("Server '{server_name}' not found in .mcp.json")).await?;
           }
           Ok(())
       }
