    2. Create `crates/rightclaw/src/codegen/cloudflared.rs`:
       ```rust
       use std::path::Path;
       use minijinja::{Environment, context};
       use serde::Serialize;

       const CF_TEMPLATE: &str = include_str!("../../../../templates/cloudflared-config.yml.j2");

       #[derive(Debug, Serialize)]
       struct CloudflaredAgent {
           name: String,
           socket_path: String,
       }

       /// Generate cloudflared ingress config YAML for OAuth callback routing.
       /// Each agent gets a path-based ingress rule routing to its Unix socket.
       /// Socket path convention: <agent_dir>/oauth-callback.sock
       pub fn generate_cloudflared_config(
           agents: &[(String, std::path::PathBuf)],  // (name, agent_dir) pairs
           tunnel_url: &str,
       ) -> miette::Result<String> {
           let cf_agents: Vec<CloudflaredAgent> = agents.iter().map(|(name, dir)| {
               CloudflaredAgent {
                   name: name.clone(),
                   socket_path: dir.join("oauth-callback.sock").to_string_lossy().to_string(),
               }
           }).collect();

           let mut env = Environment::new();
           env.add_template("cloudflared", CF_TEMPLATE)
               .map_err(|e| miette::miette!("template parse: {e:#}"))?;
           let tmpl = env.get_template("cloudflared")
               .map_err(|e| miette::miette!("template get: {e:#}"))?;
           tmpl.render(context! { agents => cf_agents, tunnel_url => tunnel_url })
               .map_err(|e| miette::miette!("template render: {e:#}"))
       }
       ```
       Per D-04: redirect_uri = `https://<tunnel-url>/oauth/<agent-name>/callback`
       Per D-09: socket path = `<agent_dir>/oauth-callback.sock`
