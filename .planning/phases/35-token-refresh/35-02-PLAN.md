```rust
use std::path::PathBuf;
use std::time::{Duration, SystemTime, UNIX_EPOCH};

use tokio::time::{sleep_until, Instant};
use tracing::{error, warn};

use crate::mcp::credentials::{read_credential, write_credential, CredentialToken};
use crate::mcp::detect::{mcp_auth_status, AuthState};
use crate::mcp::oauth::{discover_as, TokenResponse};

const REFRESH_BUFFER_SECS: u64 = 600;  // 10 minutes before expiry
const RETRY_INTERVAL_SECS: u64 = 300;  // 5 minute retry on failure
const MAX_RETRIES: u32 = 3;

/// Errors from the refresh grant flow.
#[derive(Debug, thiserror::Error)]
pub enum RefreshError {
    #[error("HTTP request failed: {0}")]
    HttpFailed(String),
    #[error("token endpoint returned error status {status}: {body}")]
    TokenEndpointError { status: u16, body: String },
    #[error("token endpoint response is not valid JSON: {0}")]
    InvalidJson(String),
    #[error("OAuth AS discovery failed: {0}")]
    DiscoveryFailed(String),
    #[error("credential read/write failed: {0}")]
    CredentialError(String),
    #[error("no refresh_token stored for server {0} — cannot refresh")]
    NoRefreshToken(String),
    #[error("no client_id stored for server {0} — cannot refresh")]
    NoClientId(String),
}
```

Implement `deadline_from_unix(expires_at_secs: u64, buffer_secs: u64) -> Option<Instant>`:
- If `expires_at_secs == 0` → return None (REFRESH-04: non-expiring)
- If `expires_at_secs <= buffer_secs` → return None (underflow guard)
- Compute `now_unix` from `SystemTime::now().duration_since(UNIX_EPOCH).unwrap_or_default().as_secs()`
- If `expires_at_secs <= now_unix` → return None (already expired — refresh immediately)
- `target_unix = expires_at_secs - buffer_secs`
- `secs_from_now = target_unix.saturating_sub(now_unix)`
- Return `Some(Instant::now() + Duration::from_secs(secs_from_now))`

Implement `async fn post_refresh_grant(client: &reqwest::Client, token_endpoint: &str, refresh_token: &str, client_id: &str, client_secret: Option<&str>) -> Result<TokenResponse, RefreshError>`:
- Build params vec: `[("grant_type", "refresh_token"), ("refresh_token", refresh_token), ("client_id", client_id)]`
- If `client_secret.is_some()`: push `("client_secret", secret)` — use owned String for lifetime
- POST to `token_endpoint` with `.form(&params)` (mirrors exchange_token in oauth.rs)
- On HTTP error: return `Err(RefreshError::HttpFailed(format!("{e:#}")))`
- If status is not 2xx: read body as text, return `Err(RefreshError::TokenEndpointError { status: status.as_u16(), body })`
- Deserialize body as `TokenResponse` with `resp.json::<TokenResponse>().await`
- On JSON error: return `Err(RefreshError::InvalidJson(format!("{e:#}")))`

Implement `pub async fn refresh_token_for_server(http_client: &reqwest::Client, credentials_path: &PathBuf, server_name: &str, server_url: &str) -> Result<u64, RefreshError>`:
- Read current credential: `read_credential(credentials_path, server_name, server_url)` — on err: return RefreshError::CredentialError
- If credential is None: return RefreshError::NoRefreshToken(server_name.to_string())
- Extract `refresh_token`: if None, return `Err(RefreshError::NoRefreshToken(server_name.to_string()))`
- Extract `client_id`: if None, return `Err(RefreshError::NoClientId(server_name.to_string()))`
- Run `discover_as(http_client, server_url).await` — on err: return `Err(RefreshError::DiscoveryFailed(format!("{e:#}")))`
- Call `post_refresh_grant(http_client, &as_meta.token_endpoint, &refresh_token, &client_id, credential.client_secret.as_deref()).await`
- Compute new `expires_at`: `now_unix + token_resp.expires_in.unwrap_or(0)` (both in seconds)
- Build new `CredentialToken`:
  ```rust
  let new_token = CredentialToken {
      access_token: token_resp.access_token,
      refresh_token: token_resp.refresh_token.or(Some(credential.refresh_token.unwrap())),
      token_type: token_resp.token_type.or(credential.token_type),
      scope: token_resp.scope.or(credential.scope),
      expires_at: new_expires_at,
      client_id: credential.client_id,      // preserve stored client_id
      client_secret: credential.client_secret, // preserve stored client_secret
  };
  ```
  NOTE: if the provider returned a new `refresh_token`, use it; otherwise keep the old one. This handles both rotating and non-rotating providers.
- Write: `write_credential(credentials_path, server_name, server_url, &new_token)` — on err: return RefreshError::CredentialError
- Return `Ok(new_expires_at)`

Implement `async fn run_token_refresh_loop(credentials_path: PathBuf, server_name: String, server_url: String, http_client: reqwest::Client)`:
- `let mut retries = 0u32;`
- Loop:
  1. Read current token (re-read each iteration — may have been updated by OAuth flow)
  2. REFRESH-04 guard: `if token.expires_at == 0 { return; }` — non-expiring, stop loop
  3. Also skip if no refresh_token: `if token.refresh_token.is_none() { warn!(server=%server_name, "no refresh_token — skipping refresh loop"); return; }`
  4. Compute deadline: `deadline_from_unix(token.expires_at, REFRESH_BUFFER_SECS)`
  5. If `Some(deadline)`: `sleep_until(deadline).await`
  6. Call `refresh_token_for_server(&http_client, &credentials_path, &server_name, &server_url).await`
  7. On Ok(new_expires_at): reset `retries = 0`; if `new_expires_at == 0 { return; }` (provider issued non-expiring replacement)
  8. On Err(e): `warn!(server=%server_name, retries, "token refresh failed: {e:#}"); retries += 1;`
     - If `retries >= MAX_RETRIES`: `error!(server=%server_name, "token refresh failed {MAX_RETRIES} times — stopping scheduler"); return;`
     - Else: `tokio::time::sleep(Duration::from_secs(RETRY_INTERVAL_SECS)).await`

Implement `pub async fn run_refresh_scheduler(agent_dir: PathBuf, credentials_path: PathBuf, http_client: reqwest::Client)`:
- `let mcp_path = agent_dir.join(".mcp.json");`
- Call `mcp_auth_status(&mcp_path, &credentials_path)` — on Err: `warn!("refresh scheduler: failed to read MCP status: {e:#}"); return;`
- For each status in results:
  - Skip if `status.state == AuthState::Missing` (no token stored)
  - Read credential to check for refresh_token presence: if `refresh_token.is_none()`: `warn!(server=%status.name, "no refresh_token stored — scheduler not started"); continue;`
  - Skip if `expires_at == 0` (non-expiring per REFRESH-04): log at debug level, not warn
  - Clone paths, spawn: `tokio::spawn(run_token_refresh_loop(credentials_path.clone(), status.name.clone(), status.url.clone(), http_client.clone()));`
